<pm>::= "+" | "-"
<MM>::= <M>"*"<MM> | <M>      // MM - Множники
<M>::= <Frac> | <Func> | <Num> | <Sqrt> | <Power> | <Var> | (<P>)       //M - множник
<Frac>::= "\frac{"<T>"}{"<T>"}"
<Func>::= "\"<FuncName>"("<P>")" | "\"<FuncName> <MM>
<FuncName>::= "sin" | "cos" | "tan" | "ctg" | "arcsin" | "arccos" | "arctan" | "arcctg"
<Num>::= [0-9]<Num>
<Sqrt>::= "\sqrt{"<P>"}" |" \sqrt["<P>"]{"<P>"}"
<Power>::= <M>"^{"<P>"}" | <M>"^"[0-9] | <M>"^"[a-zA-Z]
<Var>::=" \vec{"<VarWithPrime>"}" | <VarWithPrime>
<VarWithPrime>::= <VarWithInd><primes> | <VarWithInd>
<primes>::= '<primes> | '
